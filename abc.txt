	var initARIA = function(searchInput) {

			var acResultList = A.all('.aui-autocomplete-results .aui-autocomplete-list-item:visible');
			var searchForm = searchInput.ancestor('form');
			
			if(acResultList &amp;&amp; acResultList.size() &gt; 0) {

					A.one(searchForm).one('span[role="status"]').setContent(acResultList.size() + " Suggestion is available. use up and down arrow keys to navigate.");

				else {
					A.one(searchForm).one('span[role="status"]').setContent(acResultList.size() + " Suggestions are available. use up and down arrow keys to navigate.");
				}
			}

				A.one(searchForm).one('span[role="status"]').setContent("No Suggestions are available.");
			}
		}
	};

				}

					node.setAttribute('aria-hidden', 'false');
				}
			});
